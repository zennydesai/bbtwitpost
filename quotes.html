<body>

  <h1 id="category-title">Quotes</h1>


  <!-- Container for Quotes -->

  <div id="quotes-container"></div>


  <script>

    // Quotes Data

    const quotesData = {

      motivational: [

        "The only way to do great work is to love what you do. - Steve Jobs",

        "Believe you can and you're halfway there. - Theodore Roosevelt"

      ],

      success: [

        "Success is not final, failure is not fatal: It is the courage to continue that counts. - Winston Churchill",

        "Don’t be afraid to give up the good to go for the great. - John D. Rockefeller"

      ],

      life: [

        "In the middle of every difficulty lies opportunity. - Albert Einstein",

        "Life is what happens when you’re busy making other plans. - John Lennon"

      ]

    };


    const socialMediaIds = {

      id1: "@24_medico",

      id2: "@getivermectins"

    };


    // Function to copy text to clipboard

    function copyText(text) {

      navigator.clipboard.writeText(text).then(() => {

        alert("Copied to clipboard!");

      }).catch(err => {

        console.error("Failed to copy text: ", err);

      });

    }


    // Get category from URL

    const urlParams = new URLSearchParams(window.location.search);

    const category = urlParams.get("category");


    // Display quotes for the selected category

    const quotesContainer = document.getElementById("quotes-container");

    const categoryTitle = document.getElementById("category-title");


    if (category && quotesData[category]) {

      categoryTitle.textContent = `${category.charAt(0).toUpperCase() + category.slice(1)} Quotes`;


      quotesData[category].forEach(quote => {

        const quoteHTML = `

          <div class="quote">

            <p>${quote}</p>

            <div class="buttons">

              <button class="copy-btn" onclick="copyText('${quote} ${socialMediaIds.id1}')">Copy with ${socialMediaIds.id1}</button>

              <button class="copy-btn" onclick="copyText('${quote} ${socialMediaIds.id2}')">Copy with ${socialMediaIds.id2}</button>

            </div>

          </div>

        `;

        quotesContainer.innerHTML += quoteHTML;

      });

    } else {

      categoryTitle.textContent = "Category Not Found";

      quotesContainer.innerHTML = "<p>No quotes available for this category.</p>";

    }

  </script>
